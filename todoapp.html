App.js


                import React from 'react';
import ToDoItem from './ToDoItem'
import todos from "./todosData"

class App extends React.Component
{
    constructor(){
        super()
        this.state = {items: todos}
         this.handleChange = this.handleChange.bind(this)   
        }
    
    handleChange(id){
        this.setState(prevState => {
            const updatedTodos = prevState.items.map(tod => {
                if(tod.id === id){
                    return{
                        ...tod,
                        completed: !tod.completed 
                    }
                }
                return tod
            })
            return {items:updatedTodos}
        })
    }
    render(){
        const todoArray = this.state.items.map(todo => 
        <ToDoItem
        key = {todo.id}
        id = {todo.id} 
        item = {todo.item}
        completed = {todo.completed}
        handleChange = {this.handleChange}
        />)
        
        return(
            <div>
                {todoArray}
            </div>
        )
    }
}
export default App

TodoItem.js


    import React from 'react'

    function ToDoItem(props){
    const completedStyle = {
        fontstyle: "italic",
        textDecoration: "line-through"
    }
    return(
        <div className = "theList">
            <input type = "checkbox" checked={props.completed} onChange = {() => props.handleChange(props.id)}/>
            <p className = "listitems" style = {props.completed ? completedStyle:null }>{props.item}

        </div>
    )
}

export default ToDoItem
